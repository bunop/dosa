    DOsa

    Python wrapper for Digital Ocean API V2

        pip install dosa

        import dosa

    API_KEY = 'Your API Key'

    client = dosa.Client(api_key=API_KEY)
    client.droplets.list()
    client.images.list()
    client.droplets.create(name='terminator', region='nyc2', size='512mb', \
        image='ubuntu-14-04-x32')
